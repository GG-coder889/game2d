1 /**
  2  * @class 精灵
  3  * @augments Node
  4  * @param {String} fileName 文件路径名
  5  * @example var sprite = new Sprite({
  6  *		fileName:'sprite.png'
  7  * });
  8  */
  9 var Sprite = Class.create(Node,
 10   /** @lends Sprite# */
 11   {
 12     /**
 13 	* @description 传入图片路径初始化精灵
 14 	* @param {String} fileName 文件路径
 15 	*/
 16     initWithFile: function() {},
 17 	/**
 18 	* @description 传入帧动画初始化精灵
 19 	* @param {SpriteFrame} spriteFrame 帧动画对象
 20 	*/
 21     initWithSpriteFrame: function() {},
 22 	/**
 23 	* @description 传入纹理初始化精灵
 24 	* @param {Texture2D} texture 纹理对象
 25 	*/
 26     initWithTexture: function() {},
 27   }
 28 );
 29 
 30 Object.extend(Sprite,
 31   /** @lends Sprite */
 32   {
 33     /** 
 34 	* @description 传入图片路径创建精灵
 35 	* @function 
 36 	* @param {String} fileName 文件路径
 37 	* @param {Rect} [rect] 尺寸
 38 	* @return {Sprite} 返回精灵
 39 	* @example var sprite = spriteWithFile({
 40 	*			fileName:'sprite.png'});
 41 	* or
 42 	* var sprite = spriteWithFile({
 43 	*			fileName:'sprite.png',
 44 	*			rect : new Rect({
 45 	*				x:0,
 46 	*				y:0,
 47 	*				width:100,
 48 	*				height:100	
 49 	*			})
 50 	*});
 51 	*/ 
 52     spriteWithFile: function({fileName:fileName}) {},
 53 	/** 
 54 	* @description 传入帧动画创建精灵
 55 	* @function 
 56 	* @param {SpriteFrame} spriteFrame 帧动画对象
 57 	* @example var sprite = spriteWithSpriteFrame({
 58 	*			frame: frameObj});
 59 	* @return {Sprite} 返回精灵
 60 	*/ 
 61     spriteWithSpriteFrame: function({frame:spriteFrame}) {},
 62 	
 63 	/** 
 64 	* @description 传入纹理创建精灵
 65 	* @function 
 66 	* @param {Texture2D} texture 纹理对象
 67 	* @param {Rect} [rect] 尺寸
 68 	* @example var sprite = spriteWithTexture({
 69 	*			texture: textureObj});
 70 	* or
 71 	* var sprite = spriteWithFile({
 72 	*			texture:textureObj,
 73 	*			rect : new Rect({
 74 	*				x:0,
 75 	*				y:0,
 76 	*				width:100,
 77 	*				height:100	
 78 	*			})
 79 	*});
 80 	* @return {Sprite} 返回精灵
 81 	*/
 82     spriteWithTexture: function({texture:texture}) {},
 83   }
 84 );
