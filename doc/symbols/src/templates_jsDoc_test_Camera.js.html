 12 /** 
 13 * @description 打开手机摄像头
 14 * @param {String} quality 设置拍照图片的质量
 15 * @param {Function} callbackFunction 页面注册的回调函数，当拍照完时调用该函数，返回图片的base64值和高宽，返回值示例：{"base64":"sdf2334|dsfssdfsdfs343","width":"40",height:"40"}
 16 */ 
 17 Camera.open = function(quality, callbackFunction) {
 18 	var arguments = new Array();
 19 	arguments[0] = quality;
 20 	arguments[1] = callbackFunction;
 21 	this.execute("Camera", "open", arguments.join("|&^"));
 22 }
