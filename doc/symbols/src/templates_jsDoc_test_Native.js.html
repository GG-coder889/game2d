 62 /** 
 63 * @description 显示提示框
 64 * @param {String} title 标题
 65 * @param {String} msg 内容
 66 */ 
 67 Native.showMask = function(title,msg) {
 68 	if(typeof(title) == "undefined"){
 69 		title = "";
 70 	}
 71 	if(typeof(msg) == "undefined"){
 72 		msg = "加载中请稍后...";
 73 	}
 74 	var arguments = new Array();
 75     arguments[0] = title;
 76     arguments[1] = msg;
 77 	this.execute("Native", "showMask", arguments.join("|&^"));
 78 }
