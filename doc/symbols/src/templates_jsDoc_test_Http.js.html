 12 /** 
 13 * @description Http请求 
 14 * @param {String} method 请求方法"POST" or "GET"
 15 * @param {Boolean} async 是否使用异步
 16 * @param {Array} keyParams 上传的参数的key数组
 17 * @param {Array} valueParams 上传的参数的value数组
 18 * @param {String} url 请求的url
 19 * @param {Function} callbackFunction 页面注册的回调函数,返回示例值:adbcddd
 20 * @param {Object} returnParams 返回时调用的js callback function带入的参数
 21 */ 
 22 Http.connect = function(method, async, keyParams, valueParams, url, callbackFunction, returnParams) {
 23     var arguments = new Array();
 24     arguments[0] = method;
 25     arguments[1] = async;
 26     arguments[2] = keyParams.join("*&*");
 27     arguments[3] = valueParams.join("*&*");
 28     arguments[4] = url;
 29     arguments[5] = callbackFunction;
 30     arguments[6] = returnParams;
 31     this.execute("Http", "connect", arguments.join("|&^"));
 32 }
