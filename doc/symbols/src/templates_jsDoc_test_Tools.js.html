 12 /** 
 13 * @description 获取图片的Base64值,图片格式为jpg 
 14 * @param {String} data 图片的byte[]
 15 * @param {String} quality 图片显示质量 1 - 100
 16 * @param {Function} callbackFunction 页面注册的回调函数,返回示例值,abcdefg
 17 */ 
 18 Tools.getPicBase64Byte = function(data, quality, callbackFunction) {
 19     var arguments = new Array();
 20     arguments[0] = data;
 21     arguments[1] = quality;
 22     arguments[2] = callbackFunction;
 23     this.execute("Tools", "getPicBase64Byte", arguments.join("|&^"));
 24 }
