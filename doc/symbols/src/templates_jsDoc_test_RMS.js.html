 12 /** 
 13 * @description 将value保存到对应的key中，key是唯一的，如果已存在会覆盖 
 14 * @param {String} key 键
 15 * @param {String} value 值
 16 * @param {Function} callbackFunction 页面注册的回调函数,返回示例值,true / false
 17 */ 
 18 RMS.save = function(key, value, callbackFunction) {
 19     var arguments = new Array();
 20     arguments[0] = key;
 21     arguments[1] = value;
 22 	arguments[2] = callbackFunction;
 23     this.execute("RMS", "save", arguments.join("|&^"));
 24 }
 25 /** 
 26 * @description 读取指定key的值 
 27 * @param {String} key 键
 28 * @param {Function} callbackFunction 页面注册的回调函数,返回示例值,abcdefg
 29 */ 
 30 RMS.load = function(key, callbackFunction) {
 31     var arguments = new Array();
 32     arguments[0] = key;
 33     arguments[1] = callbackFunction;
 34     this.execute("RMS", "load", arguments.join("|&^"));
 35 }
 36 /** 
 37 * @description 删除指定key的值
 38 * @param {String} key 键
 39 */ 
 40 RMS.remove = function(key) {
 41     var arguments = new Array();
 42     arguments[0] = key;
 43     this.execute("RMS", "remove", arguments.join("|&^"));
 44 }
