  6 /**

  9 * @class 传感器 
 10 * @since version 1.0
 11 */
 12 /** 
 13 * @description 为当前页面注册一个传感器 
 14 * @param {String} type 传感器类型（传入的值是：1（加速度计传感器），2（磁力计传感器），3（方位传感器），4（陀螺仪传感器），5（光照传感器），6（压力传感器），7（温度传感器），8（距离传感器）） 
 15 * @param {function} callbackFunction 页面注册的回调函数，当传感器发生变化时调用该函数，返回相应传感器的参考值,返回值示例：{"x":"12312.123","y":"12312.123","z":"12312.123"}
 16 * @param {String} freq 采样间隔(毫秒)
 17 */ 
 18 Sensor.start = function(type, callbackFunction, freq) {
 19 	var arguments = new Array();
 20 	arguments[0] = type;
 21 	arguments[1] = callbackFunction;
 22 	arguments[2] = freq;
 23 	this.execute("Sensor", "start", arguments.join("|&^"));
 24 }
 25 /** 
 26 * @description 不使用的时候注销传感器
 27 * @param {String} type 传感器类型（传入的值是：1（加速度计传感器），2（磁力计传感器），3（方位传感器），4（陀螺仪传感器），5（光照传感器），6（压力传感器），7（温度传感器），8（距离传感器）） 
 28 */ 
 29 Sensor.stop = function(type) {
 30 	var arguments = new Array();
 31 	arguments[0] = type;
 32 	this.execute("Sensor", "stop", arguments.join("|&^"));
 33 }
