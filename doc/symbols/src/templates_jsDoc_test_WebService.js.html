 12 /** 
 13 * @description WebService请求解析 
 14 * @param {String} requestUrl 请求的url
 15 * @param {String} methodName 方法名
 16 * @param {String} keyParams key数组
 17 * @param {String} valueParams value数组
 18 * @param {Function} callbackFunction 页面注册的回调函数,返回示例值:adbcddd
 19 */ 
 20 WebService.getResult = function(requestUrl, methodName ,keyParams, valueParams, callbackFunction) {
 21     var arguments = new Array();
 22     arguments[0] = requestUrl;
 23     arguments[1] = methodName;
 24     arguments[2] = keyParams.join("*&*");
 25     arguments[3] = valueParams.join("*&*");
 26     arguments[4] = callbackFunction;
 27     this.execute("WebService", "getResult", arguments.join("|&^"));
 28 }
